    <script src="https://stackpath.bootstrapcdn.com/bootstrap/4.1.3/js/bootstrap.min.js"></script>

    <script>
    $(document).ready(function(){
      var titles = new Bloodhound({
        datumTokenizer: Bloodhound.tokenizers.whitespace,
        queryTokenizer: Bloodhound.tokenizers.whitespace,
        remote: {
          url: '/search.php?country=us&query=%QUERY',
          wildcard: '%QUERY'
        }
      });
      $("input.typeahead").typeahead(null, {
        name: 'search',
        display: 'title',
        source: titles,
        templates: {
          suggestion: function(data){
            return '<a href="/title/' + data.id + '/">' + data.title + '</a>';
          }
        }
      });
      $(window).on("hashchange", function(){
        if(location.hash === '#submitContainer'){
          window.scrollBy(0, -10);
        }
        else if(location.hash === '#filterContainer'){
          window.scrollBy(0, -63);
        }
      }).trigger("hashchange");
      var pathname = window.location.pathname;
      $(".navbar-nav > li > a[href=\"" + pathname + "\"]").parent().addClass("active");
      $(document).click(function(event){
        if(event.isTrigger){
          return;
        }
        if(!$(".search-form").is(event.target) && $(".search-form").has(event.target).length == 0 && !$(".search-input").hasClass("d-none")){
          $(".search-button").click();
        }
        if(!$(".navbar").is(event.target) && ($(".navbar").has(event.target).length == 0 || $(".search-form").has(event.target).length > 0) && $(".navbar-collapse").hasClass("collapse show")){
          $(".navbar-toggler").click();
        }
      });
      $(".search-button").click(function(event){
        if(event.isTrigger || $(".search-input").hasClass("d-none") || $(".tt-input").val() == ''){
          event.preventDefault();
        }
        else{
          return;
        }
        $(".search-input").toggleClass("d-none");
        $(".search-button").toggleClass("search-button-expanded");
        if(!$(".search-input").hasClass("d-none")){
                    $(".search-input").css("width", "auto");
                    $(".search-input").focus();
        }
        else{
          $(this).blur();
        }
      });
            $(".read-more").click(function(){
        $(this).toggleClass("read-more");
        if($(this).hasClass("read-more")){
          $(this).text("Read more...");
        }
        else{
          $(this).text("Read less...");
        }
      });
                              var rating = 0;
      $("a", ".plugin").click(function(event){
        event.preventDefault();
        var url = $(this).prop("href");
        window.open(url, "_blank");
      });
      $("span", ".rating").mouseenter(function(){
        $(this).nextAll().removeClass("fas").addClass("far text-secondary");
        $(this).prevAll().addBack().removeClass("far text-secondary").addClass("fas rate");
      });
      $(".rating").mouseleave(function(){
        $(".rated").removeClass("far text-secondary").addClass("fas");
        $("span", ".rating").not(".rated").removeClass("fas rate").addClass("far text-secondary");
      });
                        $(".list").click(function(event){
        sessionStorage.setItem('list', '1');
      });
      $("span", ".rating").click(function(){
        sessionStorage.setItem('rate', $(this).data("rating"));
      });
          });
        </script>
